SELECT 
  x,
	( z / LAG(z) OVER (ORDER BY x) - 1 ) * 100 as y,
	z
FROM (
  SELECT 
    DATE_FORMAT(date_created, '%M %Y') AS x, 
    COUNT(*) AS z
  FROM 
    dp_edd_customers
	WHERE date_created > '1970-01-01'
  GROUP BY 
    DATE_FORMAT(date_created, '%M %Y')
) subq
ORDER BY 
  STR_TO_DATE(x, '%M %Y');
